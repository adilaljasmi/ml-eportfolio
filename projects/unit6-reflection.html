# Clean feature names for better display
def clean_feature_name(name):
    """Clean up feature names for executive presentation"""
    name = str(name)
    if 'room_type_' in name:
        return 'Room Type: ' + name.replace('room_type_', '').replace('_', ' ').title()
    elif 'neighbourhood_group_' in name:
        return 'Borough: ' + name.replace('neighbourhood_group_', '')
    elif 'neighbourhood_' in name and 'neighbourhood_group' not in name:
        return 'Area: ' + name.replace('neighbourhood_', '').replace('_', ' ').title()
    elif name == 'latitude':
        return 'Location: Latitude'
    elif name == 'longitude':
        return 'Location: Longitude'
    elif name == 'availability_365':
        return 'Availability (Days/Year)'
    elif name == 'minimum_nights':
        return 'Minimum Nights'
    elif name == 'calculated_host_listings_count':
        return 'Host Listings Count'
    elif name == 'reviews_per_month':
        return 'Reviews per Month'
    elif name == 'number_of_reviews':
        return 'Number of Reviews'
    elif name == 'last_review':
        return 'Last Review (Recency)'
    else:
        return name.replace('_', ' ').title()
